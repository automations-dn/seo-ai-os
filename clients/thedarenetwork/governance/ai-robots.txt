# AI Crawler Governance for Dare Network
# ==========================================
# This file contains AI-specific robots.txt directives to control how
# AI training models (OpenAI, Anthropic, Google, etc.) crawl your site.
#
# IMPORTANT: Merge these directives into your main robots.txt file at:
# https://thedarenetwork.com/robots.txt
#
# Last Updated: 2026-03-27

# =============================================================================
# STRATEGY: Selective Access with Priority Content
# =============================================================================
# We ALLOW crawling of high-value content (case studies, service pages, blog)
# We BLOCK low-value pages (thank you pages, staging, admin, drafts)
# This ensures AI models train on our best work and cite authoritative pages.

# =============================================================================
# OpenAI (GPTBot) - ChatGPT Training & Search
# =============================================================================
User-agent: GPTBot
Allow: /case-studies                    # HIGH PRIORITY - Showcase expertise
Allow: /performance-marketing           # Service pages
Allow: /search-engine-optimization      # Service pages
Allow: /web-development                 # Service pages
Allow: /blogs/                          # Thought leadership
Allow: /about                           # Company info
Allow: /contact                         # Contact info
Allow: /llms.txt                        # AI governance feed

Disallow: /thank-you                    # No value for training
Disallow: /checkout                     # Private checkout flows
Disallow: /cart                         # E-commerce states
Disallow: /admin                        # Admin pages
Disallow: /wp-admin                     # WordPress admin
Disallow: /staging                      # Staging environment
Disallow: /?s=                          # Search results pages
Disallow: /*?utm_                       # Tracking parameters
Disallow: /drafts                       # Unpublished content
